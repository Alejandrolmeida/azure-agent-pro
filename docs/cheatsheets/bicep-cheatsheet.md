```bicep
@description('Resource prefix')
param prefix string = 'contoso'

@description('Environment')
@allowed(['dev', 'test', 'prod'])

var locationAbbreviations = {

  westus: 'wus'
  centralus: 'cus'
  eastus2: 'eus2'
  westeurope: 'weu'

  subnet: '${prefix}-${environment}-${locationAbbr}-subnet'
```
